CLASS Showtime
    PRIVATE
        showtimeID : Number
        startTime : Time
        endTime : Time
        film : Film
        room : ScreeningRoom
        tickets : List<Ticket>

    CONSTRUCTOR(showtimeID, startTime, endTime, film, room)
        SET showtimeID
        SET startTime
        SET endTime
        SET film
        SET room
        CREATE empty tickets list
    END CONSTRUCTOR

    METHOD checkOverlap(otherShowtime)
        IF room = otherShowtime.room AND
           startTime < otherShowtime.endTime AND
           endTime > otherShowtime.startTime THEN
            RETURN TRUE
        ELSE
            RETURN FALSE
        ENDIF
    END METHOD

    METHOD bookSeat(ticket)
        FOR EACH bookedTicket IN tickets
            IF bookedTicket.seat = ticket.seat THEN
                RETURN "Seat Already Booked"
            ENDIF
        END FOR

        IF SIZE(tickets) < room.getCapacity() THEN
            ADD ticket TO tickets
            RETURN "Booking Successful"
        ELSE
            RETURN "Room Full"
        ENDIF
    END METHOD

    METHOD getAvailableSeats()
        RETURN room.getCapacity() - SIZE(tickets)
    END METHOD
END CLASS
END

CLASS Policy
    PRIVATE
        policyID : Number
        issueDate : Date
        policyHolder : PolicyHolder
        coverages : List<Coverage>
        claims : List<Claim>

    CONSTRUCTOR(policyID, issueDate, policyHolder)
        SET policyID
        SET issueDate
        SET policyHolder
        CREATE empty coverages list
        CREATE empty claims list
    END CONSTRUCTOR

    METHOD addCoverage(coverage)
        ADD coverage TO coverages
    END METHOD

    METHOD fileClaim(claim)
        IF claim.getAmount() <= claim.coverage.getCoverageLimit() THEN
            ADD claim TO claims
        ELSE
            PRINT "Claim exceeds coverage limit"
        END IF
    END METHOD

    METHOD calculateRemainingCoverage()
        totalCoverage TO 0
        totalClaims TO 0

        FOR EACH coverage IN coverages
            totalCoverage ← totalCoverage + coverage.getCoverageLimit()
        END FOR

        FOR EACH claim IN claims
            IF claim.status = "Approved" THEN
                totalClaims TO totalClaims + claim.getAmount()
            END IF
        END FOR

        RETURN totalCoverage - totalClaims
    END METHOD

    METHOD getClaims()
        RETURN claims
    END METHOD
END CLASS

CLASS Event
    PRIVATE
        eventID : Number
        eventDate : Date
        budget : Number
        couple : Couple
        tasks : List<PlanningTask>
        guests : List<Guest>

    CONSTRUCTOR(eventID, eventDate, budget, couple)
        SET eventID
        SET eventDate
        SET budget
        SET couple
        CREATE empty tasks list
        CREATE empty guests list
    END CONSTRUCTOR

    METHOD addTask(task)
        IF calculateTotalCost() + task.getCost() <= budget THEN
            ADD task TO tasks
        ELSE
            PRINT "Budget Exceeded"
        END IF
    END METHOD

    METHOD addGuest(guest)
        ADD guest TO guests
    END METHOD

    METHOD calculateTotalCost()
        total to 0
        FOR EACH task IN tasks
            total to total + task.getCost()
        END FOR
        RETURN total
    END METHOD

    METHOD calculateCompletion()
        completed to 0

        FOR EACH task IN tasks
            IF task.completed = TRUE THEN
                completed ← completed + 1
            END IF
        END FOR

        RETURN (completed * 100) / SIZE(tasks)
    END METHOD
END CLASS
